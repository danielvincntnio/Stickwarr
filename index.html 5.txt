        <h2 id="detail-title" class="text-4xl font-black uppercase mb-8 tracking-tighter"></h2>
        <div id="unit-list" class="grid grid-cols-1 md:grid-cols-3 gap-6"></div>
    </div>

    <script>
        const data = {
            'Swordwrath': ['Elite Sword', 'Ninja Stick', 'Ronin Shadow'],
            'Archidon': ['Hunter Bow', 'Sniper Stick', 'Crossbow Master'],
            'Spearton': ['Iron Guard', 'Royal Shield', 'Spartan Legend']
        };

        function showDetail(type) {
            document.getElementById('home').classList.remove('active');
            document.getElementById('detail').classList.add('active');
            document.getElementById('detail-title').innerText = type;
            
            const list = document.getElementById('unit-list');
            list.innerHTML = data[type].map(unit => `
                <div class="bg-white p-6 border-2 border-gray-200 text-center hover:border-black transition-all group" onclick="triggerAnimation(this)">
                    <div class="text-5xl mb-4">👤</div>
                    <p class="font-bold mb-4 tracking-tight">${unit.toUpperCase()}</p>
                    <button onclick="window.location.href='https://stickwarr.netlify.app/'" class="w-full bg-white border-2 border-black py-2 text-[10px] font-bold uppercase tracking-widest hover:bg-black hover:text-white transition-all">Pilih</button>
                </div>
            `).join('');
        }

        function triggerAnimation(element) {
            element.classList.add('animate-click');
            setTimeout(() => element.classList.remove('animate-click'), 300);
        }

        function goHome() {
            document.getElementById('detail').classList.remove('active');
            document.getElementById('home').classList.add('active');
        }
    </script>
